function think () {
	clear_weapons();
}

function clear_weapons () {
	//Remove blacklisted weapons

	//Scouts's scatter guns (no shortstop)
	DoEntFire("tf_weapon_scattergun", "kill", "", 0, self, self);

	//Soldier's rocket launchers
	DoEntFire("tf_weapon_rocketlauncher", "kill", "", 0, self, self);
	DoEntFire("tf_weapon_rocketlauncher_directhit",	"kill", "", 0, self, self);

	//Pyro's flamethrowers (and dragons fury)
	DoEntFire("tf_weapon_flamethrower", "kill", "", 0, self, self);
	DoEntFire("tf_weapon_rocketlauncher_fireball", "kill", "", 0, self, self);

	//Demoman's sticky bomb launchers
	DoEntFire("tf_weapon_pipebomblauncher", "kill", "", 0, self, self);

	//Heavy's miniguns
	DoEntFire("tf_weapon_minigun", "kill", "", 0, self, self);

	//Engineer's PDA & related
	DoEntFire("tf_weapon_builder", "kill", "", 0, self, self);
	DoEntFire("tf_weapon_objectselection", "kill", "", 0, self, self);
	DoEntFire("tf_weapon_pda_engineer_build", "kill", "", 0, self, self);
	DoEntFire("tf_weapon_pda_engineer_destroy", "kill", "", 0, self, self);

	//Engineer's wrangler
	DoEntFire("tf_weapon_laser_pointer", "kill", "", 0, self, self);
	
	//Sniper's rifles (No bow)
	DoEntFire("tf_weapon_sniperrifle", "kill", "", 0, self, self);

	//Medic's medigun
	DoEntFire("tf_weapon_medigun", "kill", "", 0, self, self);

	//Spy's sapper
	DoEntFire("obj_attachment_sapper", "kill", "", 0, self, self);
}
